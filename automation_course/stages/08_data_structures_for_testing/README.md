    print(f"‚úÖ –í—ã–ø–æ–ª–Ω–µ–Ω–æ —Ç–µ—Å—Ç–æ–≤: {len(executed_tests)}")
    return executed_tests

def benchmark_queue_performance():
    """–°—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ –æ—á–µ—Ä–µ–¥–µ–π"""
    
    # –ë–æ–ª—å—à–æ–π –Ω–∞–±–æ—Ä –¥–∞–Ω–Ω—ã—Ö
    test_names = [f"test_{i:04d}" for i in range(10000)]
    
    # –¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ list –∫–∞–∫ –æ—á–µ—Ä–µ–¥—å
    list_queue = list(test_names)
    start_time = time.time()
    
    while list_queue:
        list_queue.pop(0)  # –ú–µ–¥–ª–µ–Ω–Ω–∞—è –æ–ø–µ—Ä–∞—Ü–∏—è –¥–ª—è –±–æ–ª—å—à–∏—Ö —Å–ø–∏—Å–∫–æ–≤
    
    list_time = time.time() - start_time
    
    # –¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ deque
    deque_queue = deque(test_names)
    start_time = time.time()
    
    while deque_queue:
        deque_queue.popleft()  # –ë—ã—Å—Ç—Ä–∞—è –æ–ø–µ—Ä–∞—Ü–∏—è O(1)
    
    deque_time = time.time() - start_time
    
    print(f"List queue time: {list_time:.4f} —Å–µ–∫")
    print(f"Deque queue time: {deque_time:.4f} —Å–µ–∫")
    print(f"Deque –±—ã—Å—Ç—Ä–µ–µ –≤ {list_time/deque_time:.1f} —Ä–∞–∑")
    
    return list_time, deque_time

# Priority queue –¥–ª—è —Ç–µ—Å—Ç–æ–≤
def priority_test_execution():
    """–û—á–µ—Ä–µ–¥—å –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è —Ç–µ—Å—Ç–æ–≤ —Å –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç–∞–º–∏"""

    print("üìã –û—á–µ—Ä–µ–¥—å —Ç–µ—Å—Ç–æ–≤ –ø–æ –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç–∞–º:")
